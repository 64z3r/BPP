 23class StandardizeFeatures(nn.Module):
 24    """Standardize features in geometric data object."""
 25
 26    def __init__(
 27        self,
 28        stats_path: str,
 29    ) -> None:
 30        """
 31        Arguments:
 32            stats_path: YAML file containing mean and standard deviation for
 33                features that should be standardized.
 34        """
 35
 36        super().__init__()
 37
 38        with open(stats_path, "r") as yf:
 39            feature_stats = yaml.safe_load(yf)
 40
 41        self.stats = nn.ParameterDict(
 42            {
 43                name: nn.ParameterList(
 44                    [
 45                        nn.Parameter(
 46                            torch.tensor(stats["mean"]),
 47                            requires_grad=False,
 48                        ),
 49                        nn.Parameter(
 50                            torch.tensor(stats["std"]),
 51                            requires_grad=False,
 52                        ),
 53                    ]
 54                )
 55                for name, stats in feature_stats.items()
 56            }
 57        )
 58
 59    def forward(self, data: Data) -> Data:
 60        """Standardize features.
 61
 62        Arguments:
 63            data: Geometric data object.
 64
 65        Returns:
 66            Geometric data object with standardized features.
 67        """
 68
 69        for name, (mean, std) in self.stats.items():
 70            x = getattr(data, name)
 71            x -= mean
 72            x /= std + 1e-9
 73            setattr(data, name, x)
 74
 75        return data

529                if not std.isnan():
530                    norms[f"weight_stat_std/{name}"] = std
531                if not min.isnan():
532                    norms[f"weight_stat_min/{name}"] = min
533                if not max.isnan():
534                    norms[f"weight_stat_max/{name}"] = max
535            self.log_dict(norms)
536
537        if "pos_coef" in self.log_net_stats:
538            norms = {}
539            for name, p in self.named_parameters():
540                if name.endswith("pos_scope") or name.endswith("pos_scale"):
541                    norms[f"pos_coef/{name}"] = p.data
542            self.log_dict(norms)
