 84def add_beta_carbon_vector(
 85    g: nx.Graph, scale: bool = True, reverse: bool = False
 86):
 87    """Adds vector from node (typically alpha carbon) to position of beta
 88    carbon.
 89
 90    Glycine does not have a beta carbon, so we set it to
 91    ``np.array([0., 0., 0.])``. We extract the position of the beta carbon from the
 92    unprocessed atomic PDB dataframe. For this we use the ``raw_pdb_df``
 93    DataFrame. If ``scale``, we scale the vector to the unit vector. If
 94    ``reverse`` is ``True``, we reverse the vector (``C beta - node``).
 95    If ``reverse`` is ``False`` (default) we compute (``node - C beta``).
 96
 97    :param g: Graph to add vector to.
 98    :type g: nx.Graph
 99    :param scale: Scale vector to unit vector. Defaults to ``True``.
100    :type scale: bool
101    :param reverse: Reverse vector. Defaults to ``False``.
102    :type reverse: bool
103    """
104    # Get or compute R-Group DF
105    if "rgroup_df" not in g.graph.keys():
106        g.graph["rgroup_df"] = compute_rgroup_dataframe(g.graph["raw_pdb_df"])
107
108    c_beta_coords = filter_dataframe(
109        g.graph["rgroup_df"], "atom_name", ["CB"], boolean=True
110    )
111    c_beta_coords.index = c_beta_coords["node_id"]
112
113    # Iterate over nodes and compute vector
114    for n, d in g.nodes(data=True):
115        if d["residue_name"] == "GLY":
116            vec = np.array([0.0, 0.0, 0.0])
117        elif n not in c_beta_coords.index:
118            vec = np.array([0.0, 0.0, 0.0])
119            log.warning(
120                f"Non-glycine residue {n} does not have a beta-carbon."
121            )
122        else:
123            if reverse:
124                vec = d["coords"] - np.array(
125                    c_beta_coords.loc[n][["x_coord", "y_coord", "z_coord"]],
126                    dtype=float,
127                )
128            else:
129                vec = (
130                    np.array(
131                        c_beta_coords.loc[n][
132                            ["x_coord", "y_coord", "z_coord"]
133                        ],
134                        dtype=float,
135                    )
136                    - d["coords"]
137                )
138
139            if scale:
140                vec = vec / np.linalg.norm(vec)
141        d["c_beta_vector"] = vec

144def add_sequence_neighbour_vector(
145    g: nx.Graph, scale: bool = True, reverse: bool = False, n_to_c: bool = True
146):
147    """Computes vector from node to adjacent node in sequence.
148    Typically used with ``CA`` (alpha carbon) graphs.
149
150    If ``n_to_c`` is ``True`` (default), we compute the vectors from the N
151    terminus to the C terminus (canonical direction). If ``reverse`` is
152    ``False`` (default), we compute ``Node_i - Node_{i+1}``. If ``reverse is
153    ``True``, we compute ``Node_{i+1} - Node_i``.
154
155    :param g: Graph to add vector to.
156    :type g: nx.Graph
157    :param scale: Scale vector to unit vector. Defaults to ``True``.
158    :type scale: bool
159    :param reverse: Reverse vector. Defaults to ``False``.
160    :type reverse: bool
161    :param n_to_c: Compute vector from N to C or C to N. Defaults to ``True``.
162    :type n_to_c: bool
163    """
164    suffix = "n_to_c" if n_to_c else "c_to_n"
165    # Iterate over every chain
166    for chain_id in g.graph["chain_ids"]:
167        # Find chain residues
168        chain_residues = [
169            (n, v) for n, v in g.nodes(data=True) if v["chain_id"] == chain_id
170        ]
171
172        if not n_to_c:
173            chain_residues.reverse()
174
175        # Iterate over every residue in chain
176        for i, residue in enumerate(chain_residues):
177            # Checks not at chain terminus - is this versatile enough?
178            if i == len(chain_residues) - 1:
179                residue[1][f"sequence_neighbour_vector_{suffix}"] = np.array(
180                    [0.0, 0.0, 0.0]
181                )
182                continue
183

185            ins_current = (
186                residue[0].split(":")[3] if residue[0].count(":") > 2 else ""
187            )

189                chain_residues[i + 1][0].split(":")[3]
190                if chain_residues[i + 1][0].count(":") > 2
191                else ""
192            )
193            if not n_to_c:
194                ins_current, ins_next = ins_next, ins_current
195

197            dist = abs(
198                residue[1]["residue_number"]
199                - chain_residues[i + 1][1]["residue_number"]
200            )
201
202            # Asserts residues are adjacent
203            cond_adjacent = (
204                dist == 1
205                or (dist == 0 and not ins_current and ins_next == "A")
206                or (
207                    dist == 0
208                    and ins_current
209                    and ins_next
210                    and chr(ord(ins_current) + 1) == ins_next
211                )
212            )
213

215            if cond_adjacent:
216                vec = chain_residues[i + 1][1]["coords"] - residue[1]["coords"]
217
218                if reverse:
219                    vec = -vec
220                if scale:
221                    vec = vec / np.linalg.norm(vec)
222            else:
223                vec = np.array([0.0, 0.0, 0.0])
224
225            residue[1][f"sequence_neighbour_vector_{suffix}"] = vec
